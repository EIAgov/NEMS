!CCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCC
!     ODBDEF:
!     CONTAINS OGSM DATABASE PARAMATER AND SWITCH VALUES                        
!CCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCC

       COMMON /ODBDEF/     OCOLVALS,  OCOLV,                         & !REAL * 4
       ORCLOGSM, OFNRUN, &
       OTNUM,     OLOOPING,  ONUMCOLS,                                 & 
       OCHCOLVALS, OCHCOLV, ODYNSTM, OSQLSTMTOLD
!
       INTEGER OMAXTABS,OMAXCOLS,OMAXRECS
       PARAMETER(OMAXTABS=20)             ! Maximum DB Tables written in a subroutine
       PARAMETER(OMAXCOLS=75)             ! Maximum Columns written in a DB record
       PARAMETER(OMAXRECS=100)           ! Maximum DB records written in one call to Load_data
!
!     STORE ORACLE SWITCH VALUES
!
      INTEGER*4    ORCLOGSM                     ! WRITE OGSM DATABASE INFO, 0=NO,1=YES
!
!     STORE FINAL RUN SWITCH
!
      INTEGER*4    OFNRUN                        ! FINAL RUN SWITCH, 1 = final run
!                                                !                   0 = not final run
!     STORE ORACLE RECORD WRITING VARIABLES
!
      INTEGER OTNUM,OLOOPING(OMAXTABS),ONUMCOLS(OMAXTABS)
      REAL*8 OCOLVALS(OMAXCOLS,OMAXRECS),OCOLV(OMAXTABS,OMAXCOLS,OMAXRECS)
      CHARACTER*25 OCHCOLVALS(OMAXCOLS,OMAXRECS),OCHCOLV(OMAXTABS,OMAXCOLS,OMAXRECS)
      CHARACTER*220 ODYNSTM(OMAXTABS)
      CHARACTER*220 OSQLSTMTOLD
!
